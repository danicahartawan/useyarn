import { motion } from 'framer-motion';

const Book = () => {
  return (
    <div style={{ perspective: '1000px' }}>
      <motion.div
        initial={{ rotateY: 0 }}
        whileHover={{ rotateY: -160 }}
        transition={{ duration: 1.5 }}
        style={{
          width: '200px',
          height: '300px',
          backgroundColor: 'navy',
          transformOrigin: 'left', // Sets the spine as the anchor
          transformStyle: 'preserve-3d',
        }}

      </motion.div>

3. Key Concepts for 3D Books
Transform Origin: Set transform-origin: left (or right for RTL languages) to ensure the page rotates around the "spine" rather than its center.
Perspective: Wrap your book in a container with a perspective property (e.g., 1000px) to give the rotation a 3D sense of depth.
Preserve-3D: Use transform-style: preserve-3d on the parent element to allow child elements (like the back of a page) to exist in the same 3D space.
